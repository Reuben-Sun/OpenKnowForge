from madmom.features.beats import RNNBeatProcessor, CRFBeatDetectionProcessor
from madmom.features.downbeats import RNNBarProcessor

def get_beats(input_audio, threshold_value=0.37):
    print(f'Process {input_audio}')
    proc = CRFBeatDetectionProcessor(fps=100)
    act = RNNBeatProcessor()(input_audio)
    data = proc(act)	# 非常密的节奏点

    proc2 = RNNBarProcessor()
